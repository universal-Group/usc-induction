    <script>
        let currentSection = 0;
        
        function showInteractiveForms() {
            document.getElementById('interactiveForms').classList.remove('hidden');
            document.querySelector('.main-content').style.display = 'none';
            updateProgress();
        }
        
        function nextSection() {
            // Validate current section
            const form = document.getElementById('employeeForm');
            const requiredFields = form.querySelectorAll('[required]');
            let isValid = true;
            
            requiredFields.forEach(field => {
                if (!field.value.trim()) {
                    field.style.borderColor = '#dc3545';
                    isValid = false;
                } else {
                    field.style.borderColor = '#e9ecef';
                }
            });
            
            if (!isValid) {
                showMessage('Please fill in all required fields', 'warning');
                return;
            }
            
            document.querySelector('.interactive-section').classList.add('hidden');
            document.getElementById('consentSection').classList.remove('hidden');
            document.getElementById('consentSection').classList.add('fade-in');
            currentSection = 1;
            updateProgress();
        }
        
        function previousSection() {
            document.getElementById('consentSection').classList.add('hidden');
            document.querySelector('.interactive-section').classList.remove('hidden');
            currentSection = 0;
            updateProgress();
        }
        
        function submitForms() {
            // Validate signature and date
            const signature = document.getElementById('signature').value.trim();
            const signatureDate = document.getElementById('signatureDate').value;
            
            if (!signature || !signatureDate) {
                showMessage('Please provide your digital signature and date', 'warning');
                return;
            }
            
            // Set today's date if not set
            if (!signatureDate) {
                document.getElementById('signatureDate').value = new Date().toISOString().split('T')[0];
            }
            
            // Simulate form submission
            setTimeout(() => {
                document.getElementById('consentSection').classList.add('hidden');
                document.getElementById('successMessage').classList.remove('hidden');
                document.getElementById('successMessage').classList.add('fade-in');
                currentSection = 2;
                updateProgress();
            }, 1000);
        }
        
        function updateProgress() {
            const progress = (currentSection / 2) * 100;
            document.getElementById('progressBar').style.width = progress + '%';
        }
        
        function showMessage(message, type) {
            const existingMessage = document.querySelector('.status-message:not(.hidden)');
            if (existingMessage) existingMessage.remove();
            
            const messageDiv = document.createElement('div');
            messageDiv.className = `status-message status-${type}`;
            messageDiv.textContent = message;
            
            document.getElementById('consentSection').insertBefore(messageDiv, document.getElementById('consentSection').firstChild);
            
            setTimeout(() => messageDiv.remove(), 5000);
        }
        
        function downloadPDF() {
            // Simulate PDF download
            alert('PDF download would start here.\n\nIn your implementation:\n1. Upload your 17-page PDF to GitHub\n2. Link to: https://yourusername.github.io/repository/your-document.pdf');
        }
        
        function generateCompletePDF() {
            // Collect all form data
            const formData = {
                firstName: document.getElementById('firstName').value,
                lastName: document.getElementById('lastName').value,
                email: document.getElementById('email').value,
                department: document.getElementById('department').value,
                startDate: document.getElementById('startDate').value,
                signature: document.getElementById('signature').value,
                signatureDate: document.getElementById('signatureDate').value,
                consents: Array.from(document.querySelectorAll('input[name="consents"]:checked')).map(cb => cb.value)
            };
            
            // In a real implementation, this would generate a PDF with the form data
            alert(`Completed forms would be generated as PDF.\n\nForm Data Collected:\n- Name: ${formData.firstName} ${formData.lastName}\n- Email: ${formData.email}\n- Department: ${formData.department}\n- Consents: ${formData.consents.length} items\n- Signed: ${formData.signature} on ${formData.signatureDate}`);
        }
        
        // Set today's date as default
        document.getElementById('signatureDate').value = new Date().toISOString().split('T')[0];
        
        // Auto-save functionality (optional)
        const formInputs = document.querySelectorAll('input, select, textarea');
        formInputs.forEach(input => {
            input.addEventListener('change', () => {
                localStorage.setItem('employeeForm_' + input.name, input.value);
            });
            
            // Load saved data
            const savedValue = localStorage.getItem('employeeForm_' + input.name);
            if (savedValue) {
                input.value = savedValue;
            }
        });
